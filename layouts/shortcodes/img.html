{{- $src := .Get "src" -}} {{- $alt := .Get "alt" | default "" -}} {{- $class := .Get "class" | default "" -}} {{- $id := .Get "id" | default "" -}} {{- $style := .Get "style" | default "" -}} {{- $priority := .Get "priority" | default "" -}} {{- $isGif := strings.HasSuffix (lower $src) ".gif" -}} {{- $img := .Page.Resources.GetMatch $src -}} {{- if not $img -}} {{- $img = resources.Get $src -}} {{- end -}} {{- if and $img (not $isGif) -}} {{- $sizes := slice 400 800 1200 1600 -}} {{- $variants := slice -}} {{- $full := $img -}} {{- range $sizes -}} {{- if gt $img.Width . -}} {{- $resized := $img.Resize (printf "%dx webp" .) -}} {{- $variants = $variants | append $resized -}} {{- end -}} {{- end -}} {{- if gt (len $variants) 0 -}} {{- $full = index $variants (sub (len $variants) 1) -}} {{- else -}} {{- $full = $img.Resize (printf "%dx webp" $img.Width) -}} {{- $variants = $variants | append $full -}} {{- end -}} {{- $srcsetParts := slice -}} {{- range $variants -}} {{- $srcsetParts = $srcsetParts | append (printf "%s %dw" .RelPermalink .Width) -}} {{- end -}} {{- $srcset := delimit $srcsetParts ", " -}} {{- if eq $priority "high" -}} {{- $lqip := $img.Resize "20x webp" -}} {{- $lqipB64 := $lqip.Content | base64Encode -}} {{- else -}} {{- end -}} {{- else if $img -}} {{/* GIF — no processing */}} {{- else -}} {{/* Fallback for images not in page bundle */}} {{- end -}}
