{{- $src := .Destination -}} {{- $alt := .Text -}} {{- $isGif := strings.HasSuffix (lower $src) ".gif" -}} {{- $img := .Page.Resources.GetMatch $src -}} {{- if and $img (not $isGif) -}} {{- $full := $img -}} {{- $small := $img -}} {{- if gt $img.Width 800 -}} {{- $full = $img.Resize "800x webp" -}} {{- else -}} {{- $full = $img.Resize (printf "%dx webp" $img.Width) -}} {{- end -}} {{- if gt $img.Width 400 -}} {{- $small = $img.Resize "400x webp" -}} {{- end -}} {{- else if $img -}} {{/* GIF — no processing, just lazy load */}} {{- else -}} {{/* Fallback for images not in page bundle (e.g. /image/...) */}} {{- end -}}
